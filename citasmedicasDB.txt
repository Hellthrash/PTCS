--Tabla especialidad
CREATE TABLE especialidad(
    id INTEGER NOT NULL,
    nombre TEXT,
    PRIMARY KEY (id));

--Tabla medico
CREATE TABLE medico(
    rut INTEGER NOT NULL,
    especialidad_id INTEGER,
    nombres TEXT,
    apellidos TEXT,
    PRIMARY KEY(rut),
    FOREIGN KEY(especialidad_id) REFERENCES especialidad(id));

--cita
CREATE TABLE cita(
    id INTEGER NOT NULL,
    paciente_rut INTEGER,
    medico_rut INTEGER,
    fecha DATETIME,
    sintomas TEXT,
    diagnostico TEXT,
    recomendaciones TEXT,
    receta TEXT,
    PRIMARY KEY (id),
    FOREIGN KEY(paciente_rut) REFERENCES paciente(rut),
    FOREIGN KEY(medico_rut) REFERENCES medico(rut));
--paciente

CREATE TABLE paciente(
    rut INTEGER NOT NULL,
    nombres TEXT,
    apellidos TEXT,
    ficha_medica TEXT,
    PRIMARY KEY(rut));
